        // Simple Navigation System
        function navigateTo(pageId) {
            console.log('Navigating to:', pageId);
            
            // Hide all pages
            const pages = document.querySelectorAll('.page');
            pages.forEach(page => {
                page.classList.remove('active');
            });
            
            // Show target page
            const targetPage = document.getElementById(pageId);
            if (targetPage) {
                targetPage.classList.add('active');
                console.log('Successfully navigated to:', pageId);
            } else {
                console.error('Page not found:', pageId);
            }
        }

        // Age Validation and Dynamic Questions
        function validateAge(input) {
            const age = parseInt(input.value);
            const validation = document.getElementById('age-validation');
            const nextBtn = document.getElementById('next-btn');
            
            if (isNaN(age) || age < 18) {
                input.classList.add('invalid');
                validation.classList.add('show');
                nextBtn.disabled = true;
                return false;
            } else {
                input.classList.remove('invalid');
                validation.classList.remove('show');
                
                // Store age and show dynamic follow-up questions
                answers.age = age;
                showAgeFollowup(age);
                nextBtn.disabled = false;
                
                return true;
            }
        }

        function showAgeFollowup(age) {
            const followupContainer = document.getElementById('dynamic-followup-1');
            let ageCategory = '';
            
            if (age >= 18 && age <= 21) ageCategory = '18-21';
            else if (age >= 22 && age <= 25) ageCategory = '22-25';
            else if (age >= 26 && age <= 30) ageCategory = '26-30';
            else if (age >= 31 && age <= 35) ageCategory = '31-35';
            else if (age >= 36 && age <= 40) ageCategory = '36-40';
            else if (age >= 41) ageCategory = '41+';
            
            followupContainer.innerHTML = dynamicQuestions.age[ageCategory];
            followupContainer.classList.add('show');
        }

        // Dynamic Question System
        function selectOption(questionNum, value, element) {
            // Store answer
            answers[`q${questionNum}`] = value;
            
            // Update UI
            const options = element.parentElement.querySelectorAll('.question-option');
            options.forEach(opt => opt.classList.remove('selected'));
            element.classList.add('selected');
            element.querySelector('input[type="radio"]').checked = true;
            
            // Show dynamic follow-up questions based on selection
            showDynamicFollowup(questionNum, value);
            
            // Enable next button
            document.getElementById('next-btn').disabled = false;
        }

        function showDynamicFollowup(questionNum, value) {
            const followupContainer = document.getElementById(`dynamic-followup-${questionNum}`);
            
            let questions = '';
            
            switch(questionNum) {
                case 2: // Physical appearance
                    questions = dynamicQuestions.physical[value] || '';
                    break;
                case 3: // Education
                    questions = dynamicQuestions.education[value] || '';
                    break;
                case 4: // Career
                    questions = dynamicQuestions.career[value] || '';
                    break;
                case 5: // Family
                    questions = dynamicQuestions.family[value] || '';
                    break;
                case 6: // Social
                    questions = dynamicQuestions.social[value] || '';
                    break;
                case 7: // Personality
                    questions = dynamicQuestions.personality[value] || '';
                    break;
            }
            
            followupContainer.innerHTML = questions;
            followupContainer.classList.add('show');
        }

        function nextQuestion() {
            // Special validation for age question
            if (currentQuestion === 1) {
                const ageInput = document.getElementById('character-age');
                if (!validateAge(ageInput)) {
                    return;
                }
            }
            
            if (currentQuestion < totalQuestions) {
                // Hide current question
                document.getElementById(`question-${currentQuestion}`).classList.remove('active');
                document.getElementById(`step-${currentQuestion}`).classList.remove('active');
                document.getElementById(`step-${currentQuestion}`).classList.add('completed');
                
                // Show next question
                currentQuestion++;
                document.getElementById(`question-${currentQuestion}`).classList.add('active');
                document.getElementById(`step-${currentQuestion}`).classList.add('active');
                
                // Update progress text
                updateProgressText();
                
                // Update buttons
                document.getElementById('prev-btn').disabled = false;
                
                // Check if this is the last question
                if (currentQuestion === totalQuestions) {
                    document.getElementById('next-btn').style.display = 'none';
                    document.getElementById('proceed-btn').style.display = 'inline-block';
                }
                
                // Scroll to top
                window.scrollTo({ top: 0, behavior: 'smooth' });
            }
        }

        function previousQuestion() {
            if (currentQuestion > 1) {
                // Hide current question
                document.getElementById(`question-${currentQuestion}`).classList.remove('active');
                document.getElementById(`step-${currentQuestion}`).classList.remove('active');
                
                // Show previous question
                currentQuestion--;
                document.getElementById(`question-${currentQuestion}`).classList.add('active');
                document.getElementById(`step-${currentQuestion}`).classList.add('active');
                
                // Update progress text
                updateProgressText();
                
                // Update buttons
                document.getElementById('next-btn').style.display = 'inline-block';
                document.getElementById('proceed-btn').style.display = 'none';
                
                if (currentQuestion === 1) {
                    document.getElementById('prev-btn').disabled = true;
                }
                
                // Scroll to top
                window.scrollTo({ top: 0, behavior: 'smooth' });
            }
        }

        function proceedToEditor() {
            console.log('Proceeding to editor');
            
            // Collect all answers including dynamic follow-up questions
            const storyParameters = {
                mainAnswers: answers,
                // Collect all dynamic follow-up answers
                dynamicFollowups: collectDynamicFollowups()
            };
            
            console.log('Parameters collected:', storyParameters);
            
            // Store parameters
            localStorage.setItem('storyParameters', JSON.stringify(storyParameters));
            
            // Navigate to editor
            navigateTo('editor');
            
            // Initialize editor
            initializeEditor();
        }

        function collectDynamicFollowups() {
            const followups = {};
            
            // Collect all textarea and input values from dynamic followups
            document.querySelectorAll('.dynamic-followup input, .dynamic-followup textarea, .dynamic-followup select').forEach(element => {
                if (element.value.trim()) {
                    followups[element.id || element.placeholder] = element.value;
                }
            });
            
            return followups;
        }

        function initializeEditor() {
            console.log('Initializing editor');
            
            const storyContent = document.getElementById('story-content');
            const wordCount = document.getElementById('word-count');
            
            // Update word count
            function updateWordCount() {
                const text = storyContent.value;
                const words = text.trim().split(/\s+/).filter(word => word.length > 0).length;
                const characters = text.length;
                wordCount.textContent = `Words: ${words} | Characters: ${characters}`;
            }
            
            // Remove existing listener to avoid duplicates
            storyContent.removeEventListener('input', updateWordCount);
            storyContent.addEventListener('input', updateWordCount);
            updateWordCount();
            
            // Load saved story if exists
            const savedStory = localStorage.getItem('savedStory');
            if (savedStory) {
                const storyData = JSON.parse(savedStory);
                document.getElementById('story-title').value = storyData.title || '';
                storyContent.value = storyData.content || '';
                updateWordCount();
            }
        }

        function saveStory() {
            console.log('Saving story');
            
            const title = document.getElementById('story-title').value;
            const content = document.getElementById('story-content').value;
            
            const storyData = {
                title: title,
                content: content,
                timestamp: new Date().toISOString(),
                parameters: JSON.parse(localStorage.getItem('storyParameters') || '{}')
            };
            
            localStorage.setItem('savedStory', JSON.stringify(storyData));
            
            // Visual feedback
            const button = event.target;
            const originalText = button.textContent;
            button.textContent = '✓ Saved';
            button.style.background = 'linear-gradient(135deg, #1B5E20, #2E7D32)';
            
            setTimeout(() => {
                button.textContent = originalText;
                button.style.background = '';
            }, 2000);
        }

        function loadStory() {
            console.log('Loading story');
            
            const savedStory = localStorage.getItem('savedStory');
            if (savedStory) {
                const storyData = JSON.parse(savedStory);
                document.getElementById('story-title').value = storyData.title || '';
                document.getElementById('story-content').value = storyData.content || '';
                
                // Update word count
                const event = new Event('input');
                document.getElementById('story-content').dispatchEvent(event);
            }
        }

        function exportStory() {
            console.log('Exporting story');
            
            const title = document.getElementById('story-title').value || 'Untitled Story';
            const content = document.getElementById('story-content').value;
            
            const storyData = {
                title: title,
                content: content,
                parameters: JSON.parse(localStorage.getItem('storyParameters') || '{}'),
                exportDate: new Date().toISOString()
            };
            
            const blob = new Blob([JSON.stringify(storyData, null, 2)], { type: 'application/json' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `${title.replace(/[^a-z0-9]/gi, '_')}.json`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
        }

        function clearStory() {
            if (confirm('Are you sure you want to clear the editor? This cannot be undone.')) {
                document.getElementById('story-title').value = '';
                document.getElementById('story-content').value = '';
                
                // Update word count
                const event = new Event('input');
                document.getElementById('story-content').dispatchEvent(event);
            }
        }

        function insertPrompt(prompt) {
            const storyContent = document.getElementById('story-content');
            const currentValue = storyContent.value;
            const newValue = currentValue + (currentValue ? '\n\n' : '') + prompt;
            storyContent.value = newValue;
            
            // Update word count
            const event = new Event('input');
            storyContent.dispatchEvent(event);
            
            // Focus on textarea
            storyContent.focus();
        }

        // Keyboard shortcuts
        document.addEventListener('keydown', function(e) {
            // Ctrl/Cmd + S to save
            if ((e.ctrlKey || e.metaKey) && e.key === 's') {
                e.preventDefault();
                if (document.getElementById('editor').classList.contains('active')) {
                    saveStory();
                }
            }
            
            // Ctrl/Cmd + E to export
            if ((e.ctrlKey || e.metaKey) && e.key === 'e') {
                e.preventDefault();
                if (document.getElementById('editor').classList.contains('active')) {
                    exportStory();
                }
            }
        });

            console.log('Shadowweave Studio initialized');

        // Navigation System
        function navigateTo(pageId) {
            console.log('Navigating to:', pageId);
            
            // Hide all pages
            const pages = document.querySelectorAll('.page');
            pages.forEach(page => {
                page.classList.remove('active');
            });
            
            // Show target page
            const targetPage = document.getElementById(pageId);
            if (targetPage) {
                targetPage.classList.add('active');
                console.log('Successfully navigated to:', pageId);
            } else {
                console.error('Page not found:', pageId);
            }
        }

        // Captor Question System
        function selectCaptorOption(questionNum, value, element) {
            // Store answer
            captorAnswers[`q${questionNum}`] = value;
            
            // Update UI
            const options = element.parentElement.querySelectorAll('.captor-option');
            options.forEach(opt => opt.classList.remove('selected'));
            element.classList.add('selected');
            element.querySelector('input[type="radio"]').checked = true;
            
            // Show dynamic follow-up questions based on selection
            showCaptorDynamicFollowup(questionNum, value);
            
            // Enable next button
            document.getElementById('captor-next-btn').disabled = false;
        }

        function showCaptorDynamicFollowup(questionNum, value) {
            const followupContainer = document.getElementById(`captor-dynamic-followup-${questionNum}`);
            
            let questions = '';
            
            switch(questionNum) {
                case 1: // Operational structure
                    questions = captorDynamicQuestions.operational[value] || '';
                    break;
                case 2: // Motivation
                    questions = captorDynamicQuestions.motivation[value] || '';
                    break;
                case 3: // Background
                    questions = captorDynamicQuestions.background[value] || '';
                    break;
                case 4: // Equipment
                    questions = captorDynamicQuestions.equipment[value] || '';
                    break;
                case 5: // Violence
                    questions = captorDynamicQuestions.violence[value] || '';
                    break;
                case 6: // Psychological
                    questions = captorDynamicQuestions.psychological[value] || '';
                    break;
                case 7: // Risk
                    questions = captorDynamicQuestions.risk[value] || '';
                    break;
                case 8: // Endgame
                    questions = captorDynamicQuestions.endgame[value] || '';
                    break;
            }
            
            followupContainer.innerHTML = questions;
            followupContainer.classList.add('show');
        }

        function nextCaptorQuestion() {
            if (currentCaptorQuestion < totalCaptorQuestions) {
                // Hide current question
                document.getElementById(`captor-question-${currentCaptorQuestion}`).classList.remove('active');
                document.getElementById(`captor-step-${currentCaptorQuestion}`).classList.remove('active');
                document.getElementById(`captor-step-${currentCaptorQuestion}`).classList.add('completed');
                
                // Show next question
                currentCaptorQuestion++;
                document.getElementById(`captor-question-${currentCaptorQuestion}`).classList.add('active');
                document.getElementById(`captor-step-${currentCaptorQuestion}`).classList.add('active');
                
                // Update progress text
                updateCaptorProgressText();
                
                // Update buttons
                document.getElementById('captor-prev-btn').disabled = false;
                
                // Check if this is the last question
                if (currentCaptorQuestion === totalCaptorQuestions) {
                    document.getElementById('captor-next-btn').style.display = 'none';
                    document.getElementById('captor-proceed-btn').style.display = 'inline-block';
                }
                
                // Scroll to top
                window.scrollTo({ top: 0, behavior: 'smooth' });
            }
        }

        function previousCaptorQuestion() {
            if (currentCaptorQuestion > 1) {
                // Hide current question
                document.getElementById(`captor-question-${currentCaptorQuestion}`).classList.remove('active');
                document.getElementById(`captor-step-${currentCaptorQuestion}`).classList.remove('active');
                
                // Show previous question
                currentCaptorQuestion--;
                document.getElementById(`captor-question-${currentCaptorQuestion}`).classList.add('active');
                document.getElementById(`captor-step-${currentCaptorQuestion}`).classList.add('active');
                
                // Update progress text
                updateCaptorProgressText();
                
                // Update buttons
                document.getElementById('captor-next-btn').style.display = 'inline-block';
                document.getElementById('captor-proceed-btn').style.display = 'none';
                
                if (currentCaptorQuestion === 1) {
                    document.getElementById('captor-prev-btn').disabled = true;
                }
                
                // Scroll to top
                window.scrollTo({ top: 0, behavior: 'smooth' });
            }
        }

        function proceedToSummary() {
            console.log('Proceeding to summary');
            
            // Collect all answers including dynamic follow-up questions
            const captorParameters = {
                mainAnswers: captorAnswers,
                dynamicFollowups: collectCaptorDynamicFollowups()
            };
            
            console.log('Captor parameters collected:', captorParameters);
            
            // Store parameters
            localStorage.setItem('captorParameters', JSON.stringify(captorParameters));
            
            // Generate summary
            generateCaptorSummary(captorParameters);
            
            // Navigate to summary page
            navigateTo('captor-summary');
        }

        function collectCaptorDynamicFollowups() {
            const followups = {};
            
            // Collect all textarea and input values from dynamic followups
            document.querySelectorAll('.captor-dynamic-followup input, .captor-dynamic-followup textarea, .captor-dynamic-followup select').forEach(element => {
                if (element.value.trim()) {
                    followups[element.id || element.placeholder] = element.value;
                }
            });
            
            return followups;
        }

        function exportCaptorConfig() {
            console.log('Exporting captor configuration');
            
            const captorParameters = JSON.parse(localStorage.getItem('captorParameters') || '{}');
            
            const exportData = {
                captorConfiguration: captorParameters,
                exportDate: new Date().toISOString(),
                version: '1.0'
            };
            
            const blob = new Blob([JSON.stringify(exportData, null, 2)], { type: 'application/json' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `captor_configuration_${Date.now()}.json`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
        }

        function resetCaptorConfig() {
            if (confirm('Are you sure you want to create a new captor configuration? This will reset all current settings.')) {
                // Reset variables
                currentCaptorQuestion = 1;
                captorAnswers = {};
                
                // Reset UI
                for (let i = 1; i <= totalCaptorQuestions; i++) {
                    document.getElementById(`captor-question-${i}`).classList.remove('active');
                    document.getElementById(`captor-step-${i}`).classList.remove('active');
                    document.getElementById(`captor-step-${i}`).classList.remove('completed');
                    
                    // Clear selections
                    const options = document.querySelectorAll(`#captor-question-${i} .captor-option`);
                    options.forEach(opt => opt.classList.remove('selected'));
                    
                    // Clear radio buttons
                    const radios = document.querySelectorAll(`#captor-question-${i} input[type="radio"]`);
                    radios.forEach(radio => radio.checked = false);
                    
                    // Clear dynamic followups
                    const followupContainer = document.getElementById(`captor-dynamic-followup-${i}`);
                    followupContainer.innerHTML = '';
                    followupContainer.classList.remove('show');
                }
                
                // Show first question
                document.getElementById('captor-question-1').classList.add('active');
                document.getElementById('captor-step-1').classList.add('active');
                
                // Reset buttons
                document.getElementById('captor-prev-btn').disabled = true;
                document.getElementById('captor-next-btn').disabled = true;
                document.getElementById('captor-next-btn').style.display = 'inline-block';
                document.getElementById('captor-proceed-btn').style.display = 'none';
                
                // Update progress
                updateCaptorProgressText();
                
                // Navigate back to config page
